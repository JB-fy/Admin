package logic

import (
	dao "api/internal/model/dao/auth"
	"api/internal/service"
	"context"

	"github.com/gogf/gf/v2/database/gdb"
)

type s{TplTableNameCamelCase} struct{}

func New{TplTableNameCamelCase}() *s{TplTableNameCamelCase} {
	return &s{TplTableNameCamelCase}{}
}

func init() {
	service.Register{TplTableNameCamelCase}(New{TplTableNameCamelCase}())
}

func (logic *s{TplTableNameCamelCase}) Count(ctx context.Context, filter map[string]interface{}) (count int, err error) {
	dao{TplTableNameCamelCase} := dao.{TplTableNameCamelCase}
	joinCodeArr := []string{}
	model := dao{TplTableNameCamelCase}.Ctx(ctx)
	if len(filter) > 0 {
		model = model.Handler(dao{TplTableNameCamelCase}.ParseFilter(filter, &joinCodeArr))
	}
	if len(joinCodeArr) > 0 {
		count, err = model.Handler(dao{TplTableNameCamelCase}.ParseGroup([]string{"id"}, &joinCodeArr)).Distinct().Count(dao{TplTableNameCamelCase}.PrimaryKey())
	} else {
		count, err = model.Count()
	}
	return
}

func (logic *s{TplTableNameCamelCase}) List(ctx context.Context, filter map[string]interface{}, field []string, order [2]string, offset int, limit int) (list gdb.Result, err error) {
	dao{TplTableNameCamelCase} := dao.{TplTableNameCamelCase}
	joinCodeArr := []string{}
	model := dao{TplTableNameCamelCase}.Ctx(ctx)
	if len(field) > 0 {
		model = model.Handler(dao{TplTableNameCamelCase}.ParseField(field, &joinCodeArr))
	}
	if len(filter) > 0 {
		model = model.Handler(dao{TplTableNameCamelCase}.ParseFilter(filter, &joinCodeArr))
	}
	if len(order) > 0 {
		model = model.Handler(dao{TplTableNameCamelCase}.ParseOrder([][2]string{order}, &joinCodeArr))
	}
	if len(joinCodeArr) > 0 {
		model = model.Handler(dao{TplTableNameCamelCase}.ParseGroup([]string{"id"}, &joinCodeArr))
	}
	if limit > 0 {
		model = model.Offset(offset).Limit(limit)
	}
	list, err = model.All()
	return
}
