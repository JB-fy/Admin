    1、依赖java11
        安装java11
            dnf install java-11-openjdk
        切换版本
            alternatives --config java
        检查JAVA_HOME是否指定非11版本（可不设置，设置后不能是非11版本）
    2、解压安装包
        tar -xvf apache-cassandra-5.0.1-bin.tar.gz
    3、移动到指定目录
        mv /server/apache-cassandra-5.0.1 /server/tool/cassandra-5.0.1
    4、数据库配置
        vi /server/tool/cassandra-5.0.1/conf/cassandra.yaml
# 集群名称，确保所有节点的集群名称相同
cluster_name: 'Test Cluster'
# 种子节点列表，用于初始化集群
seed_provider:
    - class_name: org.apache.cassandra.locator.SimpleSeedProvider
        parameters:
            - seeds: "192.168.0.200:7000,192.168.0.200:7010"
# 节点用来监听其它节点的IP地址（所有节点都在同一内网中，填内网IP，否则填公网IP）
listen_address: 192.168.0.200
# 客户端连接节点的IP地址（所有节点及应用服务器都在同一内网中时，填内网IP，否则填公网IP。可用0.0.0.0同时支持公网IP和内网IP）
rpc_address: 0.0.0.0
# 节点广播给其它节点的IP地址（所有节点都在同一内网中，填内网IP，否则填公网IP。当rpc_address设置为0.0.0.0时必填）
broadcast_rpc_address: 192.168.0.200
# CQL协议端口
native_transport_port: 9042
# 节点间的通信端口
storage_port: 7000
# 节点间的通信端口（SSL加密用）
ssl_storage_port: 7001
# 开启身份验证和授权
authenticator: PasswordAuthenticator
authorizer: CassandraAuthorizer
# 虚拟节点的数量。增加此值可以提高数据分布的均匀性
num_tokens: 256
# 并发读取线程数
concurrent_reads: 32
# 并发写入线程数
concurrent_writes: 32
# 并发物化视图写入线程数。
concurrent_materialized_view_writes: 32
# 内存表刷新写入线程数。
memtable_flush_writers: 2
# 日志级别，可以设置为 DEBUG, INFO, WARN, ERROR 等。
logger: 
    org.apache.cassandra: INFO

    5、常用命令
        启动：/server/tool/cassandra-5.0.1/bin/cassandra -f -R
        关闭：/server/tool/cassandra-5.0.1/bin/nodetool drain
        登录：/server/tool/cassandra-5.0.1/bin/cqlsh 0.0.0.0 9042 -u cassandra -p cassandra
    6、登录数据库
        /server/tool/cassandra-5.0.1/bin/cqlsh 0.0.0.0 9042 -u cassandra -p cassandra
        //修改密码
        ALTER USER 'cassandra' WITH PASSWORD '密码';
        //创建一个新的超级用户
        CREATE USER '用户名' WITH PASSWORD '密码' SUPERUSER;
        //创建普通用户
        CREATE USER '用户名' WITH PASSWORD '密码';
        //创建角色
        CREATE ROLE 角色名;
        //给角色分配权限
        GRANT SELECT ON KEYSPACE 库名 TO 角色名;    
        GRANT MODIFY ON TABLE 库名.表名 TO 角色名;
            SELECT：允许用户查询数据
            MODIFY：允许用户插入、更新和删除数据
            DROP：允许用户删除键空间或表
            CREATE：允许用户创建新的键空间或表
            ALTER：允许用户修改表结构（如添加或删除列）
            AUTHORIZE：允许用户授予和撤销其他用户的权限
            DESCRIBE：允许用户查看集群元数据
            EXECUTE：允许用户执行用户定义函数（UDFs）和用户定义聚合函数（UDAs）
        //给用户分配角色
        GRANT 角色名 TO 用户名;
        //退出数据库
        EXIT;
    7、编辑自启动文件
        vi /etc/systemd/system/cassandra-5.0.1.service
[Unit]
Description=cassandra-5.0.1
After=network.target
[Service]
ExecStart=/server/tool/cassandra-5.0.1/bin/cassandra -f -R
ExecStop=/server/tool/cassandra-5.0.1/bin/nodetool drain
PrivateTmp=true
[Install]
WantedBy=multi-user.target

    8、设置自启动
        systemctl enable cassandra-5.0.1
    9、设置环境变量增加以下目录（参考linux常用命令）
        /server/tool/cassandra-5.0.1/bin
    10、服务器对外开放配置文件中设置对外使用的端口（需要重启防火墙生效）
        firewall-cmd --zone=public --add-port=9042/tcp --permanent
        firewall-cmd --zone=public --add-port=7000/tcp --permanent
