    1、创建安装目录
        mkdir -p /server/tool/php-7.4.33
    2、解压php安装
        tar -xvf php-7.4.33.tar.gz

        进入解压目录运行
            //apache模块加载方式设置
            --with-apxs2=/tool/apache/bin/apxs
            //php-fpm方式设置
            --enable-fpm

报错处理
    报错：Cannot find ldap libraries in /usr/lib.
        ln -s /usr/lib64/libldap* /usr/lib/

    报错：Package requirements (oniguruma) were not met
        wget http://mirror.stream.centos.org/9-stream/CRB/x86_64/os/Packages/oniguruma-devel-6.9.6-1.el9.5.x86_64.rpm
        yum localinstall oniguruma-devel-6.9.6-1.el9.5.x86_64.rpm

    报错：make: *** [Makefile:749:ext/openssl/openssl.lo] 错误
    原因：centos9系统自带的openssl版本太高，需要装低版本1.1.1
        wget https://www.openssl.org/source/openssl-1.1.1q.tar.gz
        yum install perl
        tar -xvf openssl-1.1.1q.tar.gz
        cd openssl-1.1.1q
        ./config
        make && make install
		echo "/usr/local/lib64" >> /etc/ld.so.conf
		ldconfig -v
        //安装完php后，备份openssl-1.1.1q，切回系统自带的openssl-3.0.5
        mv /usr/local/bin/openssl /usr/local/bin/openssl-1.1.1q
		ln -s /usr/bin/openssl /usr/local/bin/openssl

        //8.0版本弃用
        --enable-inline-optimization --enable-maintainer-zts --with-xmlrpc --with-iconv-dir
        //可增加PKG_CONFIG_PATH=:/server/tool/x1:/server/tool/x2来增加检测依赖的目录。意味着不用--with-foo-dir=DIR等类似参数
        ./configure --prefix=/server/tool/php-7.4.33 --with-config-file-path=/server/tool/php-7.4.33/etc --with-config-file-scan-dir=/server/tool/php-7.4.33/etc/php.d --with-mysqli=mysqlnd --with-pdo-mysql=mysqlnd --enable-mysqlnd --enable-fpm --enable-mysqlnd-compression-support --enable-rpath --enable-bcmath --enable-mbregex --enable-mbstring --enable-intl --enable-ftp --enable-pcntl --enable-sockets --enable-soap --enable-gd --enable-xml --enable-exif --enable-opcache --enable-calendar --enable-session --enable-posix --enable-shmop --enable-sysvsem --enable-sysvmsg --enable-sysvshm --with-sodium --with-zlib --with-mhash --with-freetype --with-jpeg --with-webp --with-gettext --with-pear --with-xsl --with-zip --with-bz2 --with-ldap=shared --with-curl --with-openssl

        //安装
        make && make install
    3、拷贝配置文件
        //拷贝php配置文件
        cp /server/php-7.4.33/php.ini-production /server/tool/php-7.4.33/etc/php.ini
        //php-fpm启动方式设置（多个版本设置多个php-fpm，如php-fpm-7.4.33，php-fpm-7.4.33）
        cp /server/php-7.4.33/sapi/fpm/init.d.php-fpm /server/tool/php-7.4.33/bin/php-fpm-7.4.33
        cp /server/tool/php-7.4.33/etc/php-fpm.conf.default /server/tool/php-7.4.33/etc/php-fpm.conf
        cp /server/tool/php-7.4.33/etc/php-fpm.d/www.conf.default /server/tool/php-7.4.33/etc/php-fpm.d/www.conf
        //安装多版本需要改监听端口，不同版本端口设置要不一样
        vi /server/tool/php-7.4.33/etc/php-fpm.d/www.conf
    4、编辑自启动文件
        vi /etc/systemd/system/php-fpm-7.4.33.service
[Unit]
Description=php-fpm-7.4.33
After=network.target
[Service]
Type=forking
PIDFile=/server/tool/php-7.4.33/var/run/php-fpm.pid
ExecStart=/server/tool/php-7.4.33/bin/php-fpm-7.4.33 start
ExecReload=/server/tool/php-7.4.33/bin/php-fpm-7.4.33 reload
ExecStop=/server/tool/php-7.4.33/bin/php-fpm-7.4.33 stop
PrivateTmp=true
[Install]
WantedBy=multi-user.target
    5、设置自启动
        chmod -R 755 /server/tool/php-7.4.33/bin/php-fpm-7.4.33
        systemctl enable php-fpm-7.4.33
    6、设置环境变量增加以下目录（参考linux常用命令）
        /server/tool/php-7.4.33/bin
    7、扩展安装
        扩展包里面执行以下步骤
        phpize
        ./configure
        make && make install
        在php.ini添加以下内容
        extension=xxxx.so


常用扩展安装
如果php默认扩展没有安装可以重新到php安装包内的ext文件夹内找到对应的扩展安装
安装redis扩展
    tar -xvf redis-5.3.7.tgz
    进入目录编译
    phpize
    ./configure --with-php-config=/server/tool/php-7.4.33/bin/php-config
    make && make install
    在php.ini添加以下内容
    extension=redis.so
安装swoole
    tar -xvf swoole-src-5.0.1.tar.gz
    进入目录编译
    phpize
    //5.0版本弃用：--enable-http2
    ./configure --with-php-config=/server/tool/php-7.4.33/bin/php-config --enable-openssl
    make && make install
    在php.ini添加以下内容
    extension=swoole.so
安装event
    tar -xvf event-3.0.8.tgz
    进入目录编译
    phpize
    ./configure --with-php-config=/server/tool/php-7.4.33/bin/php-config
    make && make install
    在php.ini添加以下内容
    extension=event.so
安装xdebug
    tar -xvf xdebug-3.1.6.tgz
    进入目录编译
    phpize
    ./configure --with-php-config=/server/tool/php-7.4.33/bin/php-config
    make && make install
    在php.ini添加以下内容
    zend_extension=xdebug.so
    xdebug.mode=debug
    xdebug.start_with_request=yes
    xdebug.client_host=127.0.0.1
    xdebug.client_port=9003
